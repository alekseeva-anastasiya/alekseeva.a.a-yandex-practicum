SELECT cour.login, 
COUNT (ord."courierId") 
FROM  "Couriers" AS cour 
INNER JOIN "Orders" AS ord ON cour.id=ord."courierId" 
GROUP BY cour.login;

SELECT track,      
CASE WHEN finished=true THEN 2
WHEN cancelled=true THEN -1           
WHEN "inDelivery"=true THEN 1
	ELSE 0          
END 
AS status
FROM "Orders";
